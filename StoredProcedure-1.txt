	v_id lab8_tab.id%type;
   	v_lname lab8_tab.lname%type;

	-- v_id := lab1_seq.NEXTVAL;
	
	BEGIN
	SELECT last_name INTO v_lname
	FROM student
	WHERE student_id IN (SELECT student_id
                     FROM enrollment
                     GROUP by student_id
                     HAVING COUNT(*)=(SELECT MAX(count(*))
                                      FROM enrollment
                                      GROUP BY student_id))
	AND LENGTH(last_name) <9;
